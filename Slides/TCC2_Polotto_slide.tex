O objetivo desse trabalho é a construção de dispositivo embarcado que pode ser acoplado entre a fonte de alimentação e a carga de teste (monofásica) capaz de calcular a potência ativa, potência aparente e o fator de potência, com base em valores amostrados de tensão e corrente na entrada.

\section{Referencial Teórico}
\subsection{Fator de Potência}
\begin{frame}
	\frametitle{Fator de Potência}
	\framesubtitle{Definição}
	\begin{figure}
		\includegraphics[scale=1.5]{problemas_premissas/figuras/fator.eps}
		\caption{Ponto de Medição do Fator de Potência.}
		\label{fig:fator}
	\end{figure}
	\begin{exampleblock}{Fórmula:}
		\begin{equation} \label{eq:fp_eq}
		FP = \frac{P}{S} = \frac{\frac{1}{T}\int_{0}^{T}v(t)i(t)dt}{V_{RMS} I_{RMS}}
		\end{equation}
	\end{exampleblock}
\end{frame}
%
\begin{frame}
	\frametitle{Fator de Potência}
	\framesubtitle{Resentação para CL}
	\begin{figure}
		\includegraphics[scale=2]{problemas_premissas/figuras/pot_fasor.eps}
		\caption{Representação Vetorial das Potências para CL.}
		\label{fig:pot_fasor}
	\end{figure}
	\begin{exampleblock}{Fórmula:}
		\begin{equation} \label{eq:fp_eq}
		FP = {\cos \varphi}
		\end{equation}
	\end{exampleblock}
\end{frame}
%
\begin{frame}
	\frametitle{Fator de Potência}
	\framesubtitle{Respresentação para CNL}
	\begin{figure}
		\includegraphics[width=0.4\textwidth,height=0.40\textheight]{problemas_premissas/figuras/pot_fasor_har.eps}
		\caption{Representação Vetorial das Potências para CNL.}
		\label{fig:pot_fasor}
	\end{figure}
	\begin{exampleblock}{Fórmula:}
		\begin{equation} \label{eq:fp_eq}
		FP = \frac{cos(\varphi)}{\sqrt{1+I_{THD}^2}} = cos(\varphi) cos(\lambda)
		\end{equation}
	\end{exampleblock}
\end{frame}
%
\begin{frame}
	\frametitle{Fator de Potência}
	\framesubtitle{Taxa de Distorção Harmônica}
	\begin{figure}
		\centering
			\includegraphics[width=0.3\textwidth,height=0.35\textheight]{problemas_premissas/figuras/thd.eps}
			\caption{Distorção Harmônica da Corrente..}
	\end{figure}
	
	\begin{exampleblock}{Definição:}
		\begin{equation}
			I_{RMS} = \sqrt{I_{1 RMS}^2+\sum_{n=2}^{\infty}I_{n RMS}^2}
		\end{equation}
	\end{exampleblock}
	
\end{frame}

\begin{frame}
	\frametitle{Fator de Potência}
	\framesubtitle{Taxa de Distorção Harmônica}
	\begin{alertblock}{Observação:}
		Para a curva de tensão, a distorção não é significativa.
	\end{alertblock}
\end{frame}

\begin{frame}
	\frametitle{Fator de Potência}
	\framesubtitle{Componentes Harmônicas}
	\begin{figure}
		\centering
			\includegraphics[width=0.4\textwidth,height=0.40\textheight]{problemas_premissas/figuras/thd_freq.eps}
			\caption{Corrente com distorção harmônica no domínio da frequência.}
	\end{figure}
	\begin{exampleblock}{Definição:}
		\begin{equation}
			I_{THD} = \frac{\sqrt{\sum_{n=2}^{\infty}I_{n RMS}^2}}{I_{1 RMS}}
		\end{equation}
	\end{exampleblock}
\end{frame}

\section{Conclusões}
\begin{frame}
	\frametitle{Conclusões}
	\justify
	\begin{itemize}
		\item O circuito das cargas de teste produziram as ondas esperadas;
		\item Ruído prejudicial à exatidão foi encontrado na corrente amostrada;
		\item CRI apresentou o maior erro percentual médio ($11,81\%$) devido aos ruídos;
		\item O medidor confundiu a classificação das cargas CRI e CRC em alguns momentos (a obtenção do $\theta$ pode ser falha);
		\item As medidas efetuadas foram todas próximas aos valores teóricos e simulados;
		\item O dispositivo funcionou como o esperado;
		\item Provou a eficiência do uso digital dos métodos descritos no Referencial Teórico;
		\item Pode ser aplicado em experimentos laboratoriais.
	\end{itemize}
\end{frame}

\section{Propostas para Trabalhos Futuros}
\begin{frame}
	\frametitle{Propostas para Trabalhos Futuros}
	\begin{itemize}
		\item Buscar novos métodos para medida de corrente que não sejam tão ruidosos;
		\item Amostrar mais períodos de rede para se obter uma medida mais exata;
		\item Remover o circuito detector de cruzamento por zero através de um \textit{software} mais inteligente;
		\item Verificar a possibilidade de utilizar o sistema como medidor digital de energia;
		\item Chaveamento automático da escala do medidor para melhorar a resolução de medida de tensão e corrente;
		\item Utilizar o dispositivo como um controlador de banco de capacitores.
	\end{itemize}
\end{frame}
